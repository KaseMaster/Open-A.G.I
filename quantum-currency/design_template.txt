    <script>
        // Global State Simulation
        let apiStatus = 'Stopped';
        let systemStatus = 'Ready';
        let autoBalance = 'Disabled';
        let currentCoherence = 0.000;
        const coherenceThreshold = 0.750;
        
        // UHES Data
        const uhesStatus = {
            calEngine: 'Inactive', memoryLayers: '0/0/0', entropyIssues: 0, autoTuning: 'Disabled',
            globalCoherence: 0.000, systemicEquilibrium: 0.000, connectedNodes: 0, caf: 0.000, lastUpdated: 'Never'
        };

        // Utility Functions
        const updateStatus = (id, value, lightClass = 'light-red') => {
            const el = document.getElementById(id);
            if (el) {
                el.textContent = value;
                const lightEl = document.getElementById(id + 'Light');
                if (lightEl) {
                    lightEl.className = lightClass;
                }
            }
        };

        const updateCoherence = (newCoherence) => {
            currentCoherence = Math.min(1.0, Math.max(0.0, newCoherence));
            const width = (currentCoherence * 100).toFixed(2);
            
            document.getElementById('coherenceValue').textContent = currentCoherence.toFixed(3);
            document.getElementById('coherenceFill').style.setProperty('--coherence-width', `${width}%`);

            uhesStatus.globalCoherence = currentCoherence;
            
            // Check threshold for system status light
            const statusLightClass = currentCoherence >= coherenceThreshold ? 'light-green' : (currentCoherence > 0.3 ? 'light-yellow' : 'light-red');
            updateStatus('systemStatus', systemStatus, statusLightClass);
            
            renderGlobalDashboard();
        };

        const logResult = (message, color = 'text-gray-300') => {
            const resultsDiv = document.getElementById('transactionResults');
            const historyDiv = document.getElementById('transactionHistory');
            const logEntry = `<p class="${color} font-mono text-xs mb-1">[${new Date().toLocaleTimeString()}] ${message}</p>`;

            // Update Transaction Panel Result
            resultsDiv.innerHTML = logEntry;
            
            // Update History Panel
            historyDiv.innerHTML = logEntry + historyDiv.innerHTML;
        };

        const handleApiToggle = () => {
            const btn = document.getElementById('apiToggleBtn');
            
            if (apiStatus === 'Stopped' || apiStatus === 'Ready') {
                apiStatus = 'Running';
                systemStatus = 'Operational';
                btn.textContent = 'Stop REST API';
                updateStatus('apiStatus', apiStatus, 'light-green');
                updateStatus('systemStatus', systemStatus, 'light-green');
                logResult('REST API started. System Operational.', 'text-emerald-400');
            } else {
                apiStatus = 'Stopped';
                systemStatus = 'Ready';
                btn.textContent = 'Start REST API';
                updateStatus('apiStatus', apiStatus, 'light-red');
                updateStatus('systemStatus', systemStatus, 'light-red');
                logResult('REST API stopped. System Ready.', 'text-red-400');
            }
        };

        const runDemo = () => {
            if (apiStatus !== 'Running') {
                handleApiToggle();
            }

            // Update UHES Status for Demo
            uhesStatus.calEngine = 'Active';
            uhesStatus.memoryLayers = '3/3/3';
            uhesStatus.autoTuning = 'Enabled';
            
            updateStatus('calEngineStatus', uhesStatus.calEngine, 'light-green');
            updateStatus('memoryLayersStatus', uhesStatus.memoryLayers, 'light-green');
            updateStatus('autoTuningStatus', uhesStatus.autoTuning, 'light-cyan');

            // Simulate Coherence increase
            let step = 0;
            const interval = setInterval(() => {
                const increase = (Math.random() * 0.1) + 0.05;
                const newCoherence = currentCoherence + increase;
                
                if (newCoherence >= 0.9) {
                    clearInterval(interval);
                    updateCoherence(0.935);
                    uhesStatus.connectedNodes = 5;
                    uhesStatus.systemicEquilibrium = 0.950;
                    uhesStatus.caf = 0.980;
                    uhesStatus.lastUpdated = new Date().toLocaleTimeString();
                    renderGlobalDashboard();
                    logResult('Demo finished: Coherence stabilized above threshold (0.935).', 'text-emerald-400');
                    return;
                }

                updateCoherence(newCoherence);
                logResult(`Coherence Attunement: ${newCoherence.toFixed(3)}`, 'text-cyan-400');
                step++;
            }, 500);
        };
        
        const renderGlobalDashboard = () => {
            document.getElementById('globalCoherenceValue').textContent = uhesStatus.globalCoherence.toFixed(3);
            document.getElementById('systemicEquilibriumValue').textContent = uhesStatus.systemicEquilibrium.toFixed(3);
            document.getElementById('connectedNodesValue').textContent = uhesStatus.connectedNodes;
            document.getElementById('cafValue').textContent = uhesStatus.caf.toFixed(3);
            document.getElementById('lastUpdatedValue').textContent = uhesStatus.lastUpdated;
            // Also update Ledger Balances (simulated)
            document.getElementById('flxBalances').innerHTML = `
                <p class="font-mono text-sm flex justify-between">SYSTEM_A: <span class="text-yellow-300">1000.00 FLX</span></p>
                <p class="font-mono text-sm flex justify-between">USER_ALPHA: <span class="text-yellow-300">500.50 FLX</span></p>
            `;
            document.getElementById('chrReputation').innerHTML = `
                <p class="font-mono text-sm flex justify-between">SYSTEM_A: <span class="text-purple-300">0.95 CHR</span></p>
                <p class="font-mono text-sm flex justify-between">USER_ALPHA: <span class="text-purple-300">0.88 CHR</span></p>
            `;
        };

        window.onload = () => {
            document.getElementById('coherenceThreshold').textContent = coherenceThreshold.toFixed(3);
            updateCoherence(currentCoherence); 
            renderGlobalDashboard();
            
            document.getElementById('apiToggleBtn').onclick = handleApiToggle;
            document.getElementById('runDemoBtn').onclick = runDemo;
            // Other button handlers would go here for a full implementation
        };

    </script>
